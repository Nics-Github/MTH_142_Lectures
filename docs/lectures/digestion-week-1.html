    • This adds a cloumn

babies_lbw <- babies |>
  mutate(bwt_type= 
           case_when(
            bwt < 3.23*16 ~ "very low birth weight",
            bwt > 3.23*16 &  bwt < 5.5*16 ~ "low birth weight",
            bwt >= 5.5*16 ~ "healthy birth weight"
    )
  )
# Numbers are multiplied by 16 to give ounces from pounds
